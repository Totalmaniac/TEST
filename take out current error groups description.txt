# List of groups
$groups = "Group1", "Group2", "Group3" 

# Connect to Exchange Online  
Connect-ExchangeOnline

# Loop through groups
foreach ($group in $groups) {

  # Get group details
  $groupDetails = Get-DistributionGroup -Identity $group

  # Output group name and current description
  New-Object -TypeName PSObject -Property @{
      "Name" = $groupDetails.DisplayName
      "Description" = $groupDetails.Description
  }  
}

# Disconnect
Disconnect-ExchangeOnline

# Groups to retrieve details for
$groups = "Group1", "Group2", "Group3"

# Import AD module for cmdlets 
Import-Module ActiveDirectory

# Loop through each group
foreach ($group in $groups) {

  # Get group details from AD
  $groupDetails = Get-ADGroup -Identity $group -Properties Description

  # Output group name and description
  New-Object -TypeName PSObject -Property @{  
      "Name" = $groupDetails.Name
      "Description" = $groupDetails.Description
  }
}
